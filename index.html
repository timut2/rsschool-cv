                import { useState } from 'react';
                function App() {
                  const [isPuzzleSolved, setIsPuzzleSolved] = useState(false); // State to control puzzle solution

                    <div>
                      {!isPuzzleSolved ? (
                        <Puzzle onSolve={ () => setIsPuzzleSolved(true)} /> // Passing function to solve puzzle

                          <div className="main-container">
                            <Main />
                            <Main />
                            <Main />
                            <Main />
                          </div>
                          <Footer />
                        <>
                      )}
                    </div>
                  );
                }
                export default App;
